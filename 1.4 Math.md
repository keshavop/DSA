class Solution {

    int reverse(int x) {
        int temp = 0;
        long long rev = 0;
        while(x!=0){
            temp = x%10;
            rev = rev *10 + temp;
            x = x/10;
        }
        if(rev>INT_MAX || rev<INT_MIN){
            return 0;
        }
    return rev;
    }

>### 3. Check Palindrome
- Leetcode
```cpp
class Solution {
public:
    bool isPalindrome(int x) {
        if(x<0 || x%10==0 && x!=0){
            return false;
        }
        long long rev = 0;
        int num = x;
        while(num!=0){
            int ld = num %10;
            rev = rev * 10 + ld;
            num = num / 10;
        }
    return rev == x;
    }
};
```

>### 4. LCM And GCD
- GFG
```cpp
    vector<long long> lcmAndGcd(long long A , long long B) {
        int minNum = min(A,B);
        long long gcd=0;
        long long lcm=0;
        for(int i =1; i<=minNum; i++){
            if(A % i ==0 && B % i==0){
                gcd = i;
            }
        }
        lcm = A*B/gcd;
        
        vector<long long> ans;
        ans.push_back(lcm);
        ans.push_back(gcd);
    return ans;
    }
```
